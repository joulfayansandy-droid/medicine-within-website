The video-loader.js expects specific data attributes to optimize performance:

<!-- Optimized Video Container -->
<div class="video-container">
    <!-- Fallback image / Poster -->
    <div class="video-poster">
        <img src="/img/poster.jpg" class="video-poster-fallback">
        <div class="video-play-btn">▶</div>
    </div>

    <!-- Video element with lazy loading data attributes -->
    <video 
        data-lazy-video 
        data-src="/videos/desktop.mp4" 
        data-src-mobile="/videos/mobile.mp4"
        data-click-to-play
        muted loop
    ></video>
</div>
